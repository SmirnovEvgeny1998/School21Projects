int s21_strlen(char* str) {
    int len = 0;
    while (*str != '\0') {
        str++;
        len++;
    }
    return len;
}

int s21_strcmp(char* str1, char* str2) {
    int result = 0;
    while (*str1 != '\0' || *str2 != '\0') {
        if (result != 0) {
            break;
        } else if (*str1 == *str2) {
            result = 0;
        } else {
            if (*str1 > *str2) {
                result = *str1 - *str2;
                if (*str1 == '\0' || *str2 == '\0') {
                    break;
                }
            } else {
                result = *str1 - *str2;
                if (*str1 == '\0' || *str2 == '\0') {
                    break;
                }
            }
        }
        str1++;
        str2++;
    }
    return result;
}
